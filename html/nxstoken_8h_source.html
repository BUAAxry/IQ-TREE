00023 /*----------------------------------------------------------------------------------------------------------------------
00024 |       NxsToken objects are used by NxsReader to extract words (tokens) from a NEXUS data file. NxsToken objects know to
00025 |       correctly skip NEXUS comments and understand NEXUS punctuation, making reading a NEXUS file as simple as repeatedly
00026 |       calling the GetNextToken() function and then interpreting the token returned. If the token object is not attached 
00027 |       to an input stream, calls to GetNextToken() will have no effect. If the token object is not attached to an output
00028 |       stream, output comments will be discarded (i.e., not output anywhere) and calls to Write or Writeln will be 
00029 |       ineffective. If input and output streams have been attached to the token object, however, tokens are read one at a
00030 |       time from the input stream, and comments are correctly read and either written to the output stream (if an output
00031 |       comment) or ignored (if not an output comment). Sequences of characters surrounded by single quotes are read in as
00032 |       single tokens. A pair of adjacent single quotes are stored as a single quote, and underscore characters are stored
00033 |       as blanks.
00034 */

00042                         parentheticalToken              = 0x0002,       /* if set, and if next character encountered is a left parenthesis, token will include everything up to the matching right parenthesis */
00043                         curlyBracketedToken             = 0x0004,       /* if set, and if next character encountered is a left curly bracket, token will include everything up to the matching right curly bracket */
00044                         doubleQuotedToken               = 0x0008,       /* if set, grabs entire phrase surrounded by double quotes */

00046                         newlineIsToken                  = 0x0020,       /* if set, newline character treated as a token and atEOL set if newline encountered */

00166 /*----------------------------------------------------------------------------------------------------------------------
00167 |       Reads next character from in and does all of the following before returning it to the calling function:
00168 |~
00169 |       o if character read is either a carriage return or line feed, the variable line is incremented by one and the
00170 |         variable col is reset to zero
00171 |       o if character read is a carriage return, and a peek at the next character to be read reveals that it is a line
00172 |         feed, then the next (line feed) character is read
00173 |       o if either a carriage return or line feed is read, the character returned to the calling function is '\n' if 
00174 |         character read is neither a carriage return nor a line feed, col is incremented by one and the character is
00175 |         returned as is to the calling function
00176 |       o in all cases, the variable filepos is updated using a call to the tellg function of istream.
00177 |~
00178 */
00179 inline char NxsToken::GetNextChar()
00180         {
00181         int ch = in.get();
00182         int failed = in.bad();
00183         if (failed)
00184                 {
00185                 errormsg = "Unknown error reading data file (check to make sure file exists)";
00186                 throw NxsException(errormsg);
00187                 }
00188 
00189         if (ch == 13 || ch == 10)
00190                 {
00191                 fileline++;
00192                 filecol = 1L;
00193 
00194                 if (ch == 13 && (int)in.peek() == 10)
00195                         ch = in.get();
00196 
00197                 atEOL = 1;
00198                 }
00199         else if (ch == EOF)
00200                 atEOF = 1;
00201         else
00202                 {
00203                 filecol++;
00204                 atEOL = 0;
00205                 }
00206 
00207 #       if defined(__DECCXX)
00208                 filepos = 0L;
00209 #       else
00210                 filepos = in.tellg();
00211 #       endif
00212 
00213         if (atEOF)
00214                 return '\0';
00215         else if (atEOL)
00216                 return '\n';
00217         else
00218                 return (char)ch;
00219         }

00221 /*----------------------------------------------------------------------------------------------------------------------
00222 |       Returns true if character supplied is considered a punctuation character. The following twenty characters are 
00223 |       considered punctuation characters:
00224 |>
00225 |       ()[]{}/\,;:=*'"`+-<>
00226 |>
00227 |       Exceptions:
00228 |~
00229 |       o The tilde character ('~') is also considered punctuation if the tildeIsPunctuation labile flag is set
00230 |       o The special punctuation character (specified using the SetSpecialPunctuationCharacter) is also considered 
00231 |         punctuation if the useSpecialPunctuation labile flag is set
00232 |       o The hyphen (i.e., minus sign) character ('-') is not considered punctuation if the hyphenNotPunctuation 
00233 |         labile flag is set
00234 |~
00235 |       Use the SetLabileFlagBit method to set one or more NxsLabileFlags flags in `labileFlags'
00236 */
00237 inline bool NxsToken::IsPunctuation(
00238   char ch)      /* the character in question */
00239         {
00240         // PAUP 4.0b10 
00241         //  o allows ]`<> inside taxon names
00242         //  o allows `<> inside taxset names
00243         //
00244         bool is_punctuation = false;
00245         if (strchr(punctuation, ch) != NULL)
00246                 is_punctuation = true;
00247         if (labileFlags & tildeIsPunctuation  && ch == '~')
00248                 is_punctuation = true;
00249         if (labileFlags & useSpecialPunctuation  && ch == special)
00250                 is_punctuation = true;
00251         if (labileFlags & hyphenNotPunctuation  && ch == '-')
00252                 is_punctuation = false;
00253 
00254         return is_punctuation;
00255         }

00258 |       Returns true if character supplied is considered a whitespace character. Note: treats '\n' as darkspace if labile
00259 |       flag newlineIsToken is in effect.
00260 */
00261 inline bool NxsToken::IsWhitespace(
00262   char ch)      /* the character in question */
00263         {
00264         bool ws = false;
00265 
00266         // If ch is found in the whitespace array, it's whitespace
00267         //

00269                 ws = true;

00271         // Unless of course ch is the newline character and we're currently
